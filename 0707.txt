hobbyの中に配列を増やした場合、echoができなくなった
配列の場合の判定をいれて、foreachで出力する様に変更した
foreach($users as $user) {
   
    echo $user['name']."\n\r";
    
    echo $user['weight']."\n\r";
    
    echo $user['height']."\n\r";
    
    foreach($user['hobbies'] as $hobby) {  
        
        if(!is_array($hobby)){
           
           echo $hobby."\n\r";
       
        } else {
             
             foreach($hobby as $okashi) {
                echo $okashi."\n\r";
             }
        }
     
    }
}
